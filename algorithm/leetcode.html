# 704.二分查找

class Solution {
  public int search(int[] nums, int target) {
      return search(nums, target, 0, nums.length - 1);
  }

  private int search(int[] nums, int target, int min, int max) {

    if (nums[min] == target) {
        return min;
    }
    if (nums[max] == target) {
      return max;
    }

    int mid = (min + max) / 2;
    int minValue = nums[mid];

    if (mid == min) {
      return minValue == target ? mid : -1;
    }

    if (minValue == target) {
      return mid;
    } else if (minValue > target) {
      return search(nums, target, min, mid);
    } else {
      return search(nums, target, mid + 1, max);
    }
  }
}

# 278.第一个错误的版本

public class Solution extends VersionControl {
  public int firstBadVersion(int n) {
      int left = 1, right = n;

      while (left < right) {        
          // 如果中间的是错误版本，
          int mid = left + (right - left) / 2; // 防止计算时溢出 如果直接使用left+right有可能会溢出
          if (isBadVersion(mid)) {
              right = mid;
          } else  {
            left = mid + 1;
          }
      }

      return left;

  }
}

# 35.搜索插入位置

class Solution {
  public int searchInsert(int[] nums, int target) {
    int left = 0, right = nums.length;

    while (left < right) {
      int mid = left +  (right - left) / 2;
      if (nums[mid] == target) {
        return mid;
      } else if (nums[mid] > target) {
        right = mid;
      } else {
        left = mid + 1;
      }
    }

    return left;
  }
}

# 977. 有序数组的平方

// 数组可能存在负值
class Solution {
  public int[] sortedSquares(int[] nums) {
    int left = 0, right = nums.length - 1;
    int[] res = new int[nums.length];
    int last = nums.length - 1;
    while (left < right) {
      if (Math.abs(nums[left]) < Math.abs(nums[right])) {
        res[last] = nums[right] * nums[right];
        right --;
      } else {
        res[last] = nums[left] * nums[left];
        left++;
      }

      last--;
    }
    res[last] = nums[left] * nums[left];
    return res;
  }
}

# 189. 旋转数组

// 此方法会超时
class Solution {
  public void rotate(int[] nums, int k) {
    k = k % nums.length;
    for (int i = 0; i < k; i++) {
      int tmp = nums[0];
      for (int j = 1; j < nums.length; j++) {
        int temp = nums[j];
        nums[j] = tmp;
        tmp = temp;
      }
      nums[0] = tmp;
    }
  }
}

class Solution {
  public void rotate(int[] nums, int k) {
    k = k % nums.length;

    int[] tmp = new int[nums.length];

    for (int i = 0; i < nums.length; i++) {
      tmp[(i + k)%(nums.length)] = nums[i];
    }

    for (int i = 0; i < nums.length; i++) {
      nums[i] = tmp[i];
    }
  }
}
