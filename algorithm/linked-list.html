# 反转整个链表

public class ReverseLinked {

	public ListNode reverse(ListNode head) {
		if (head.next == null) {
			// 识别尾节点
			return head;
		}

		ListNode last = reverse(head.next);
		head.next.next = head;
		head.next = null;
		return last;
	}
}

# 反转链表的前n个节点

public class ReverseLinked {

	ListNode successor = null;

	public ListNode reverseN(ListNode head, int n) {
			if (n == 1) {
				successor = head.next;
				return head;
			}

			ListNode last = reverseN(head.next, n - 1);
			head.next.next = head;
			head.next = successor;

			return last;
	}
}

# 反转链表的一部分


public class ReverseLinked {

	// 记录区间结尾的后驱节点
	ListNode successor = null;

	// 使用递归的方式进行反转
	ListNode reverseBetween(ListNode head, int m, int n) {

		if (m == 1) {
			// 当m = 1时，相当于反转前n个节点
			return reverseN(head, n);
		}

		// 其他的时候，递归处理
		head.next = reverseBetween(head.next, m - 1, n - 1);
		return head;
	}

	/**
	 * 反转前n个节点, 返回新的节点头
	 */
	ListNode reverseN(ListNode head, int n) {
		if (n == 1) {
			// 记录第n + 1个节点，从这个节点开始不用反转，直接拼到反转后的节点即可
			successor = head.next;
			return head;
		}

		// 以head.next为头，则需要反转前 n-1 个节点
		ListNode last = reverseN(head.next, n - 1);

		//
		head.next.next = head;

		// 让反转之后的 head 节点和后面的节点连接起来
		head.next = successor;
		return last;
	}
}

# k个一组反转链表


ListNode reverse(ListNode head) {
	ListNode pre, cur, next;
	pre = null; cur = head; next = head;

	while (cur != null) {
		next = cur.next;
		cur.next = pre;
		pre = cur;
		cur = next;
	}

	return pre;
}

// [a,b)
ListNode reverse(ListNode a, ListNode b) {
	ListNode pre, cur, next;
	pre = null;
	cur = a;
	next = a;
	while (cur != b) {
		next = cur.next;
		cur.next = pre;
		pre = cur;
		cur = next;
	}

	return pre;
}

public ListNode reverseKGroup(ListNode head, in k) {
	if (head == null) {
		return null;
	}
	ListNode a, b;
	a = b = head;

	for (int i = 0; i < k; i++) {
		if (b == null) {
			return head;
		}

		b = b.next;
	}

	ListNode newHead = reverse(a, b);
	a.next = reverseKGroup(b, k);
	return newHead;
}
